            1, Biến
                khai báo: số, chuỗi, ngày  tháng ...
                int x = 5 
                var tên biến = giá trị
                giá trị => quy định kiểu dữ liệu của biến 
                value = số ==> datatype: số 
                value = chuỗi ==> datatype: chuỗi
                var a = 100 ; // a có kiểu dữ liệu dạng số 
                var b = "dev"; // b có kiểu dữ liệu dạng chuỗi
            2, Toán tử 
                các phép số học: +, -, *, /, %
                var x = 10;
                var y = "dev";
                var z = x + y;
                => quy tắc phép cộng:
                số + số ==> số 
                chuỗi + chuỗi = chuỗi
                chuỗi + số ==> chuỗi: vd: 10 + '50' = '1050'
                console.log("giá trị của z =", z);
                console.log("giá trị của z = "+ z)
                Các phép toàn logic 
                so sánh: ==, >, <, >=, <=, ===
                điều kiện; &&, ||, !

                    if - else : switch - case

                    if(điều kiên){
                        các câu lệnh Js sẽ được thực thi nếu điều kiện là đúng ...
                    }
                    else{
                        .. các câu lệnh được thực thi nếu điều kiện là sai...
                    }
                4, for; while, do - while
                    Cú pháp 
                    for(giá trị đếm; điều kiện ; action){
                        thân for: js
                    }
                    thứ tự for:
                        b1: khởi tạo biến đếm
                        b2: check điều kiện: 
                            - điều kiện đúng: thực thi thân for => chạy hành động => quay trở lại b2
                            - điều kiện sai : dừng for/kết thúc
